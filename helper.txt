git init - инициализировали репозиторий
git status - отслеживаем статус проекта
git add index.html - добавили новый файл в будущий коммит
git add . - добавить все файлы
git rm --cached index.html - удаляем файл из будущего коммита

git branch - просмотр, в какой ветке мы находимся
git branch readme - создали новую ветку с названием readme
git checkout readme - перешли на ветку readme
git checkout -b new - перешли и создали сразу новую ветку
git branch -D new - удалили ветку new
git merge readme - соединили master с branch

git remote add origin https://github.com/semen1994i/git-course.git - заливаем на Git
git push -u origin main
